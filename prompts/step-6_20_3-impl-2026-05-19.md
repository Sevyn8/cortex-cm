**Function 1: enforce_role_audience_scope_coherence**

```sql
CREATE OR REPLACE FUNCTION core.enforce_role_audience_scope_coherence()
RETURNS TRIGGER AS $$
DECLARE
    v_role_audience TEXT;
    v_perm_scope TEXT;
BEGIN
    SELECT audience::TEXT INTO v_role_audience FROM core.roles WHERE id = NEW.role_id;
    SELECT scope::TEXT INTO v_perm_scope FROM core.permissions WHERE id = NEW.permission_id;
    IF v_role_audience = 'TENANT' AND v_perm_scope = 'GLOBAL' THEN
        RAISE EXCEPTION
            'TENANT-audience role cannot hold GLOBAL-scope permission (role_id=%, permission_id=%)',
            NEW.role_id, NEW.permission_id
            USING ERRCODE = '23514';
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tg_role_permissions_audience_scope_coherence
    BEFORE INSERT OR UPDATE OF role_id, permission_id
    ON core.role_permissions
    FOR EACH ROW
    EXECUTE FUNCTION core.enforce_role_audience_scope_coherence();
```

**Function 2: protect_super_admin_override_global_grant**

```sql
CREATE OR REPLACE FUNCTION core.protect_super_admin_override_global_grant()
RETURNS TRIGGER AS $$
DECLARE
    v_super_admin_id UUID;
    v_override_global_id UUID;
BEGIN
    SELECT id INTO v_super_admin_id FROM core.roles WHERE code = 'SUPER_ADMIN';
    SELECT id INTO v_override_global_id FROM core.permissions WHERE code = 'ADMIN.ROLES.OVERRIDE.GLOBAL';
    IF OLD.role_id = v_super_admin_id AND OLD.permission_id = v_override_global_id THEN
        RAISE EXCEPTION
            'Cannot delete SUPER_ADMIN OVERRIDE.GLOBAL grant (platform bootstrap protection)'
            USING ERRCODE = '23514';
    END IF;
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tg_role_permissions_protect_super_admin_override
    BEFORE DELETE
    ON core.role_permissions
    FOR EACH ROW
    EXECUTE FUNCTION core.protect_super_admin_override_global_grant();
```

**Function 3: protect_super_admin_role**

```sql
CREATE OR REPLACE FUNCTION core.protect_super_admin_role()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'DELETE' THEN
        IF OLD.code = 'SUPER_ADMIN' THEN
            RAISE EXCEPTION
                'SUPER_ADMIN role cannot be deleted (platform bootstrap protection)'
                USING ERRCODE = '23514';
        END IF;
        RETURN OLD;
    ELSIF TG_OP = 'UPDATE' THEN
        IF OLD.code = 'SUPER_ADMIN' AND (
            NEW.code IS DISTINCT FROM OLD.code OR
            NEW.status IS DISTINCT FROM OLD.status OR
            NEW.audience IS DISTINCT FROM OLD.audience
        ) THEN
            RAISE EXCEPTION
                'SUPER_ADMIN role status, code, and audience are immutable (platform bootstrap protection)'
                USING ERRCODE = '23514';
        END IF;
        RETURN NEW;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tg_roles_protect_super_admin
    BEFORE UPDATE OR DELETE
    ON core.roles
    FOR EACH ROW
    EXECUTE FUNCTION core.protect_super_admin_role();
```

Note for Trigger 3: trigger watches UPDATE on the whole row (not UPDATE OF specific columns) because the function needs access to OLD.code regardless of which fields changed. Pre-flight Check #6 may reveal a more elegant existing pattern; mirror if found.
